-- All published courses with category name
SELECT c.title, c.level, cat.name AS category, c.status
FROM courses c
JOIN categories cat ON c.category_id = cat.id
WHERE c.status = 'published';

-- How many students per course
SELECT c.title, COUNT(e.user_id) AS total_enrolled
FROM courses c
LEFT JOIN enrollments e ON c.id = e.course_id
GROUP BY c.id, c.title
ORDER BY total_enrolled DESC;

-- All lessons in a course with section name
SELECT s.title AS section, l.title AS lesson, l.duration
FROM lessons l
JOIN sections s ON l.section_id = s.id
WHERE l.course_id = 1
ORDER BY s.position, l.position;
